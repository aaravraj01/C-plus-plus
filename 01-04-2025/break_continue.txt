
//-----------------------------------------Break------------------------------------------------------//

 -> Break: "Jumpout"


for(i=1; i<=5; i++)
{
    if(i==3)
    {
        break;
    }
    cout<<i<<endl;
}


//-----------------------------------------Continue------------------------------------------------------//

 -> Continue: "skip"


for(i=1; i<=5; i++)
{
    if(i==3)
    {
        continue;
    }
    cout<<i<<endl;
}
